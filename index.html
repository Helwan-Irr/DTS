firebase.auth().onAuthStateChanged((user) => { if (user) { // التأكد من أن الإيميل مسموح له بالدخول في قاعدة البيانات const userEmail = user.email; const db = firebase.firestore(); db.collection("authorized_emails").doc(userEmail).get().then((doc) => { if (!doc.exists) { alert("عذراً، هذا الحساب غير مسجل في منظومة ري حلوان."); firebase.auth().signOut(); window.location.href = "login.html"; } }); } else { window.location.href = "login.html"; } });
